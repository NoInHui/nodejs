import React, { Component } from 'react';
import ReactDOM from 'react-dom';

// 엘리먼트 렌더링
class ReactOfficial2 extends Component {
    constructor(props) {
        super(props);
        // React 엘리먼트는 불변객체 입니다.
        // 엘리먼트를 생성한 이후에는 해당 엘리먼트의 자식이나 속성을 변경할 수 없습니다.
        // 지금까지 소개한 내용을 바탕으로 하면 UI 를 업데이트하는 유일한 방법은 새로운 엘리먼트를 생성하고 이를 root.render() 로 전달하는 것입니다.
        setInterval(this.tick, 1000);
    }
    tick() {
        const element = (
            <div>
                <h1>Hello, wolrd!</h1>
                <h2>It is {new Date().toLocaleTimeString()}.</h2>
            </div>
        )

        // React 로 구현된 애플리케이션은 일반적으로 하나의 루트 DOM 노드가 있습니다.
        // React 를 기존 앱에 통합하려는 경우 원하는 만큼 많은 수의 독립된 루트 DOM 노드가 있을 수 있습니다.
        // React 엘리먼트를 렌더링 하기 위해서는 우선 DOM 엘리먼트를 ReactDOM.createRoot()  에 전달한 다음, React 엘리먼트를 root.render() 에 전달해야 합니다.
        const root = ReactDOM.createRoot(
            document.querySelector('#root2')
        );

        root.render(element);
    }
    render() {
        // 엘리먼트는 React 앱의 가장 작은 단위입니다.
        const element1 = <h1>Hello, world</h1>;
        // 브라우저 DOM 엘리먼트와 달리 React 엘리먼트는 일반 객체입니다.
        // React DOM 은 React 엘리먼트와 일치하도록 DOM 을 업데이트합니다.

        // 더 널리 알려진 개념인 컴포넌트와 엘리먼트를 혼동할 수 있습니다.
        // 엘리먼트는 컴포넌트의 구성 요소입니다.

        return (
            <div id='root2'></div>
        );
    }
}

export default ReactOfficial2;
